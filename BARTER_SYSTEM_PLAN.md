```typescript
// –ê–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏–π –ø–æ–∏—Å–∫ —Å–æ–≤–ø–∞–¥–µ–Ω–∏–π
async function findWishlistMatches(userId: string): Promise<{
  exhibit: Exhibit;
  owner: string;
  matchScore: number;
}[]> {
  const wishlist = await getWishlist(userId);
  const allExhibits = await getAllExhibits();
  const matches: any[] = [];

  wishlist.items.forEach(wantedItem => {
    allExhibits.forEach(exhibit => {
      // –ù–µ –ø–æ–∫–∞–∑—ã–≤–∞—Ç—å —Å–≤–æ–∏ –∞—Ä—Ç–µ—Ñ–∞–∫—Ç—ã

      // –ü—Ä–æ–≤–µ—Ä–∏—Ç—å –∫–∞—Ç–µ–≥–æ—Ä–∏—é
      if (exhibit.category !== wantedItem.category) return;

      // –ü—Ä–æ–≤–µ—Ä–∏—Ç—å –ø–æ–¥–∫–∞—Ç–µ–≥–æ—Ä–∏—é (–µ—Å–ª–∏ —É–∫–∞–∑–∞–Ω–∞)
      if (wantedItem.subcategory &&
          exhibit.subcategory !== wantedItem.subcategory) {
        return;
      }

      // –ü—Ä–æ–≤–µ—Ä–∏—Ç—å –∫–ª—é—á–µ–≤—ã–µ —Å–ª–æ–≤–∞
      const titleLower = exhibit.title.toLowerCase();
      const descLower = exhibit.description.toLowerCase();

      let keywordMatches = 0;
      wantedItem.keywords.forEach(keyword => {
        const kw = keyword.toLowerCase();
        if (titleLower.includes(kw) || descLower.includes(kw)) {
          keywordMatches++;
        }
      });

      if (keywordMatches === 0) return;

      // –ü—Ä–æ–≤–µ—Ä–∏—Ç—å –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–µ —Å–æ—Å—Ç–æ—è–Ω–∏–µ
      if (wantedItem.minCondition) {
        const conditionRank = getConditionRank(exhibit.condition);
        const minRank = getConditionRank(wantedItem.minCondition);
        if (conditionRank < minRank) return;
      }

      // –†–∞—Å—Å—á–∏—Ç–∞—Ç—å match score
      const matchScore = (keywordMatches / wantedItem.keywords.length) * 100;

      matches.push({
        exhibit,
        owner: exhibit.owner,
        matchScore: Math.round(matchScore)
      });
    });
  });

  // –°–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ –ø–æ —Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω–æ—Å—Ç–∏
  return matches.sort((a, b) => b.matchScore - a.matchScore);
```
